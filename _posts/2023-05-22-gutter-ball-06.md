## 개요
---
모바일로 사용하는 PWA의 특성 상, 한 번 로그인하면 리프레쉬 토큰이 만료될 때까지는 <br>
새로고침 시에도 로그인 상태가 유지되어야 한다. 해당 기능을 구현해보았다.<br>

## getUserInfo
---
{% highlight jsx linenos %}
export const getUserInfo = async () => {
  try {
    const response = await axiosInstance.get('/members/profile')
    return { status: 'success', userInfo: response.data }
  } catch (error) {
    console.error(error)
    return { status: 'fail', userInfo: null }
  }
}
{% endhighlight %}

## refreshAccessToken
---
{% highlight jsx linenos %}
export const refreshAccessToken = async () => {
  try {
    const { headers } = await axiosInstance.get('/members/refresh')
    axiosInstance.defaults.headers.common.Authorization = headers.authorization
    return 'success'
  } catch (error) {
    return 'fail'
  }
}
{% endhighlight %}

## 로그인 유지 함수 로직
---
{% highlight tsx linenos %}
const authHandler = async () => {
  // 액세스 토큰을 헤더에 실어 userInfo를 요청한다.
  const { userInfo } = await getUserInfo()
  // 응답으로 돌아온 데이터가 없을 경우 (액세스 토큰이 만료됐거나 없을 경우)
  if (userInfo === null) {
    // 액세스 토큰을 refresh하는 요청을 보낸다.
    const isRefreshed = await refreshAccessToken()
    // refresh에 성공했을 경우
    if (isRefreshed === 'success') {
      // 다시 userInfo를 요청한다.
      const { userInfo } = await getUserInfo()
      // 요청에 성공한 경우
      if (userInfo) {
        // 로그인 상태를 유지하고 각각의 전역 상태를 갱신한다.
        setIsLogin(true)
        setId(userInfo.memberId)
        setUser(userInfo)
        return 'login'
      }
    // refresh에 실패했을 경우(refresh 토큰까지 만료된 경우)
    } else {
      // 로그아웃 상태로 만든다.
      setIsLogin(false)
      return 'logout'
    }
  // 최초 요청(액세스 토큰으로 userInfo 요청)이 성공했을 경우
  } else {
    // 로그인 상태를 유지하고 각각의 전역 상태를 갱신한다.
    setIsLogin(true)
    setId(userInfo.memberId)
    setUser(userInfo)
    return 'login'
  }
}
{% endhighlight %}
